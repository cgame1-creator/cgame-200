    let lv = 1, score = 0, qIdx = 0, timeLeft = 10, timerId = null, quizSet = [], currentMax = 10, lvName = "";
    const app = document.getElementById('app');
    const lvMap = { 1: "執行特工", 2: "資深特派", 3: "王牌特攻", 4: "特級首席" };

    function showLobby() {
        if(timerId) clearInterval(timerId);
        app.innerHTML = `
            <div style="font-size:70px;">🦅</div>
            <div class="main-title">成語特工：航太終端</div>
            <div style="font-size:32px; margin-bottom:30px; color:var(--cyan)">啟動 400 題任務序列：</div>
            <div class="level-grid">
                <div class="level-btn btn-l1" onclick="start(1)"><div class="lv-tag">${lvMap[1]}</div></div>
                <div class="level-btn btn-l2" onclick="start(2)"><div class="lv-tag">${lvMap[2]}</div></div>
                <div class="level-btn btn-l3" onclick="start(3)"><div class="lv-tag">${lvMap[3]}</div></div>
                <div class="level-btn btn-l4" onclick="start(4)"><div class="lv-tag">${lvMap[4]}</div></div>
            </div>
        `;
    }

    function start(n, fast = false) {
        if (audioCtx.state === 'suspended') audioCtx.resume();
        lv = n; lvName = lvMap[n]; score = 0; qIdx = 0; 
        currentMax = fast ? 7 : 10;
        // 隨機洗牌從 100 題抽 10 題
        quizSet = [...quizData[n]].sort(() => Math.random() - 0.5).slice(0, 10);
        run();
    }

    function run() {
        if (qIdx >= quizSet.length) { finish(); return; }
        if (timerId) clearInterval(timerId);
        timeLeft = currentMax;
        let d = quizSet[qIdx];
        let options = [{text: d[1], isCorrect: true}, {text: d[2], isCorrect: false}].sort(() => Math.random() - 0.5);
        app.innerHTML = `
            <div style="width:100%; display:flex; justify-content:space-between; align-items:center; padding:0 30px; box-sizing:border-box;">
                <span style="font-size:30px; color:var(--cyan)">${lvName}</span>
                <div class="timer-big" id="t">${timeLeft}</div>
                <span style="font-size:30px; color:var(--cyan)">MISSION: ${qIdx+1}/10</span>
            </div>
            <div class="q-text">「${d[0]}」的意思是？</div>
            <button class="opt-btn" id="opt0" onclick="handleCheck(0, ${options[0].isCorrect})">${options[0].text}</button>
            <button class="opt-btn" id="opt1" onclick="handleCheck(1, ${options[1].isCorrect})">${options[1].text}</button>
            <div id="expl-area"></div>
        `;
        timerId = setInterval(() => {
            timeLeft--; 
            const tEl = document.getElementById('t');
            if (tEl) { tEl.innerText = timeLeft; if (timeLeft <= 3) tEl.className = "timer-big timer-urgent"; }
            if (timeLeft <= 0) { clearInterval(timerId); handleCheck(-1, false, true); }
        }, 1000);
    }

    function handleCheck(btnIndex, isCorrect, isTimeout = false) {
        if (timerId) clearInterval(timerId); 
        const d = quizSet[qIdx];
        const explArea = document.getElementById('expl-area');
        const btn0 = document.getElementById('opt0'), btn1 = document.getElementById('opt1');
        btn0.onclick = null; btn1.onclick = null;
        if (isCorrect && !isTimeout) {
            score += 100; SFX.ding();
            document.getElementById('opt' + btnIndex).classList.add('opt-correct');
            explArea.innerHTML = `<div class="expl-panel expl-ok">⭐ 正確回報：${d[3]}</div>`;
        } else {
            SFX.fail();
            if(!isTimeout && btnIndex !== -1) document.getElementById('opt' + btnIndex).classList.add('opt-wrong');
            if(btn0.innerText === d[1]) btn0.classList.add('opt-correct'); else btn1.classList.add('opt-correct');
            const prefix = isTimeout ? "⌛ 通訊超時：" : "❌ 攔截錯誤：";
            explArea.innerHTML = `<div class="expl-panel expl-fail">${prefix}${d[3]}</div>`;
            app.animate([{transform:'translateX(-20px)'},{transform:'translateX(20px)'},{transform:'translateX(0)'}], {duration:100, iterations:3});
        }
        qIdx++; setTimeout(run, 3000);
    }

    function finish() {
        if(timerId) clearInterval(timerId);
        if (score >= 800) {
            SFX.win(); let end = Date.now() + 4000;
            (function f() { confetti({ particleCount: 20, spread: 80, origin: { y: 0.6 } }); if (Date.now() < end) requestAnimationFrame(f); })();
            app.innerHTML = `<div class="main-title">評定成功！</div><div class="score-hero">${score}</div><button class="opt-btn" onclick="start(lv, true)">⚡ 挑戰原職：加速版 (7秒)</button><button class="opt-btn" onclick="showLobby()">⭐ 返回總部晉升職銜</button>`;
        } else {
            SFX.fail();
            app.innerHTML = `<div class="main-title" style="color:var(--red)">任務失敗</div><div class="score-hero" style="color:var(--red)">${score}</div><button class="opt-btn" onclick="showLobby()">重新受訓</button>`;
        }
    }
    showLobby();
</script>
